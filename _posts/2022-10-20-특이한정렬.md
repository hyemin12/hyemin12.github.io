function solution(numlist, n) {
  var answer = [];
  const sortArr = numlist
    .map((v) => [v, Math.abs(v - n)])
    .sort((a, b) => (a[1] === b[1] ? b[0] - a[0] : a[1] - b[1]));

  answer = sortArr.map((v) => v[0]);

  return answer;
}
