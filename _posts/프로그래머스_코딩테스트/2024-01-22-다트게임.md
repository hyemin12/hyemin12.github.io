function solution(dartResult) {
  const numList = [];

  dartResult.split("").forEach((r, i) => {
    switch (r) {
      case "0": {
        if (dartResult[i - 1] === "1") {
          numList.pop();
          numList.push(10);
          break;
        }
        numList.push(0);
        break;
      }
      case "S":
        break;
      case "D": {
        const n = numList.pop();
        numList.push(n ** 2);
        break;
      }
      case "T": {
        const n = numList.pop();
        numList.push(n ** 3);
        break;
      }
      case "*": {
        const current = numList.pop();
        const prev = numList.pop();
        if (prev) numList.push(prev * 2);
        if (current) numList.push(current * 2);
        break;
      }
      case "#": {
        const prev = numList.pop();
        numList.push(prev * -1);
        break;
      }
      default:
        numList.push(+r);
    }
  });
  // ìµœì¢…ê°’ ê³„ì‚°
  const answer = numList.reduce((acc, cur) => (acc += cur), 0);
  return answer;
}
