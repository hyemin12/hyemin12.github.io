function solution(arr) {
  const length = arr.length;
  let diff;

  for (let i = 1; i <= 1000; i *= 2) {
    if (length == i) return arr;
    if (length < i) {
      diff = i - arr.length;
      break;
    }
  }

  return [...arr, ...Array(diff).fill(0)];
}
