function solution(arr) {
  var answer = [];
  const length = Math.max(arr.length, arr[0].length);

  for (let i = 0; i < length; i++) {
    arr[i]
      ? answer.push([...arr[i], ...Array(length - arr[i].length).fill(0)])
      : answer.push(Array(length).fill(0));
  }
  return answer;
}
