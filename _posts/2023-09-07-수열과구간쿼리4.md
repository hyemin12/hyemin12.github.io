function solution(arr, queries) {
  var answer = arr;

  for (query of queries) {
    const [s, e, k] = query;
    answer = answer.map((element, i) =>
      s <= i && i <= e && i % k === 0 ? (element += 1) : element
    );
  }
  return answer;
}
