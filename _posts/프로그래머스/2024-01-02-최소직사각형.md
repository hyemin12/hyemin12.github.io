function solution(sizes) {
  let maxWidth = 0;
  let maxHeight = 0;
  sizes.forEach((size) => {
    const [w, h] = size.sort((a, b) => a - b);
    maxWidth = maxWidth < w ? w : maxWidth;

  });
  return maxWidth * maxHeight;
}
