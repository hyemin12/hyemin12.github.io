```js
function solution(s) {
  const answer = [];
  const arr = s.split(" ");

  for (let i = 0; i < arr.length; i++) {
    let word = arr[i].split("");
    for (let j = 0; j < word.length; j++) {
      word[j] = j % 2 === 0 ? word[j].toUpperCase() : word[j].toLowerCase();
    }
    answer.push(word.join(""));
  }
  return answer.join(" ");
}
```
