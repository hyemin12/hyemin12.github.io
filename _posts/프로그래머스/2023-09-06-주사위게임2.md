function solution(a, b, c) {
  const calcPow = (count) => {
    var sum = 1;
    for (let i = 1; i <= count; i++) {
      sum *= Math.pow(a, i) + Math.pow(b, i) + Math.pow(c, i);
    }
    return sum;
  };

  if (a !== b && b !== c && a !== c) return a + b + c;
  if ((a === b && a !== c) || (a === c && a !== b) || (b === c && a !== b))
    return calcPow(2);
  if (a === b && a === c) return calcPow(3);
}

  const [a, b, c] = inc;

  for (let i = 1; i <= n; i++) {

  return result;
};
